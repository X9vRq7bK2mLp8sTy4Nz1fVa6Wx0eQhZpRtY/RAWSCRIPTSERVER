// Load current content from server async function loadContent() { try { const response = await fetch('/api/content.js?key=ROBLOX_SECRET_2024'); if (!response.ok) { throw new Error('Failed to fetch content'); } const content = await response.text(); document.getElementById('contentText').value = content; } catch (error) { console.error('Failed to load content:', error); document.getElementById('contentText').value = 'Error loading content. Please refresh.'; // Try alternative endpoint try { const response = await fetch('/content?key=ROBLOX_SECRET_2024'); if (response.ok) { const content = await response.text(); document.getElementById('contentText').value = content; } } catch (fallbackError) { console.error('Fallback also failed:', fallbackError); } } }
